# `numberField(): ZodField<number | undefined>`

```ts
import { numberField } from "@form-atoms/field";

// required number, must be filled to submit
const heightCentimeters = numberField({
  required_error: "Please set your height.",
});

// optional number, can be empty to submit
const tipAmount = numberField().optional();
```

## useNumberFieldProps(numberField)

A hook providing props to control a number input.

```tsx
const NumberInput = ({
  field,
  label,
}: {
  field: NumberFieldAtom;
  label: ReactNode;
}) => {
  const props = useNumberFieldProps(field);

  return (
    <div style={{ margin: "20px 0" }}>
      <FieldLabel field={field} label={label} />
      <input type="number" {...props} />
      <FieldErrors field={field} />
    </div>
  );
};
```
